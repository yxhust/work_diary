## 创建时钟类
### 一、一次性走seconds秒，秒超过60、分超过60做分支判断
```python
class Clock(object):
    def __init__(self, hour, minute, second):
        self.hour = hour
        self.minute = minute
        self.second = second

    def run(self, seconds):
        if self.second + seconds < 60:
            self.second += seconds
        else:
            delta_min = (self.second + seconds) // 60
            new_sec = (self.second + seconds) % 60
            self.second = new_sec
            if self.minute + delta_min < 60:
                self.minute += delta_min
            else:
                delta_hour = (self.minute + delta_min) // 60
                new_min = (self.minute + delta_min) % 60
                self.minute = new_min
                if self.hour + delta_hour < 24:
                    self.hour += delta_hour
                else:
                    self.hour = 0
                    print("next day")

    def show(self):
        print("%s:%s:%s" % (self.hour, self.minute, self.second))

if __name__ == "__main__":
    clock = Clock(14, 36, 21)
    clock.run(35702)
    clock.show()
```

### 二、一次性走seconds秒，但在run函数内部是按秒计算时间的。
```python
class Clock(object):
    def __init__(self, hour, minute, second):
        self.hour = hour
        self.minute = minute
        self.second = second

    def run(self, seconds):
        count = 0
        while count < seconds:
            count += 1
            if self.second + 1 < 60:
                self.second += 1
            else:
                self.second = 0
                if self.minute + 1 < 60:
                    self.minute += 1
                else:
                    self.minute = 0
                    if self.hour + 1 < 24:
                        self.hour += 1
                    else:
                        self.hour = 0

    def show(self):
        print("%s:%s:%s" % (self.hour, self.minute, self.second))

if __name__ == "__main__":
    clock = Clock(14, 36, 21)
    clock.run(3000)
    clock.show()
```

### 三、一次性走一秒，在主函数中设置一次走一秒，一秒打印一次最新时间，**推荐**
```python
import time
class Clock(object):
    def __init__(self, hour, minute, second):
        self.hour = hour
        self.minute = minute
        self.second = second

    def run(self):
        self.second += 1
        if self.second == 60:
            self.minute += 1
            self.second = 0
            if self.minute == 60:
                self.minute = 0
                self.hour += 1
                if self.hour == 24:
                    self.hour = 0

    def show(self):
        # 时间格式
        print("%02d:%02d:%02d" % (self.hour, self.minute, self.second))

def main():
    clock = Clock(14, 36, 51)
    while True:
        clock.show()
        clock.run()
        time.sleep(1)

if __name__ == "__main__":
    main()
```

## 定义一个类 描述平面上的点并提供移动点和计算到另一个点距离的方法
### 我写的类，明确是否需要输入数据即可
```python
import math
class Point(object):
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def move(self, delta_x, delta_y):
        self.x += delta_x
        self.y += delta_y

    def distance(self, a, b):
        print("与(a,b)的距离:", math.sqrt((self.x - a)**2 + (self.y - b)**2))

    def location(self):
        print("当前坐标(%d, %d)" % (self.x, self.y))

def main():
    point = Point(3, 4)
    point.location()
    point.move(3, 6)
    point.location()
    point.distance(9, 6)

if __name__ == '__main__':
    main()

```

```python
from math import sqrt


class Point(object):

    def __init__(self, x=0, y=0):
        """初始化方法
        
        :param x: 横坐标
        :param y: 纵坐标
        """
        self.x = x
        self.y = y

    def move_to(self, x, y):
        """移动到指定位置
        
        :param x: 新的横坐标
        "param y: 新的纵坐标
        """
        self.x = x
        self.y = y

    def move_by(self, dx, dy):
        """移动指定的增量
        
        :param dx: 横坐标的增量
        "param dy: 纵坐标的增量
        """
        self.x += dx
        self.y += dy

    def distance_to(self, other):
        """计算与另一个点的距离
        
        :param other: 另一个点
        """
        dx = self.x - other.x
        dy = self.y - other.y
        return sqrt(dx ** 2 + dy ** 2)

    def __str__(self):
        return '(%s, %s)' % (str(self.x), str(self.y))


def main():
    p1 = Point(3, 5)
    p2 = Point()
    print(p1)
    print(p2)
    p2.move_by(-1, 2)
    print(p2)
    print(p1.distance_to(p2))


if __name__ == '__main__':
    main()
```
