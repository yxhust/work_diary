## 创建时钟类
### 一、一次性走seconds秒，秒超过60、分超过60做分支判断
```python
class Clock(object):
    def __init__(self, hour, minute, second):
        self.hour = hour
        self.minute = minute
        self.second = second

    def run(self, seconds):
        if self.second + seconds < 60:
            self.second += seconds
        else:
            delta_min = (self.second + seconds) // 60
            new_sec = (self.second + seconds) % 60
            self.second = new_sec
            if self.minute + delta_min < 60:
                self.minute += delta_min
            else:
                delta_hour = (self.minute + delta_min) // 60
                new_min = (self.minute + delta_min) % 60
                self.minute = new_min
                if self.hour + delta_hour < 24:
                    self.hour += delta_hour
                else:
                    self.hour = 0
                    print("next day")

    def show(self):
        print("%s:%s:%s" % (self.hour, self.minute, self.second))

if __name__ == "__main__":
    clock = Clock(14, 36, 21)
    clock.run(35702)
    clock.show()
```

### 二、一次性走seconds秒，但在run函数内部是按秒计算时间的。
```python
class Clock(object):
    def __init__(self, hour, minute, second):
        self.hour = hour
        self.minute = minute
        self.second = second

    def run(self, seconds):
        count = 0
        while count < seconds:
            count += 1
            if self.second + 1 < 60:
                self.second += 1
            else:
                self.second = 0
                if self.minute + 1 < 60:
                    self.minute += 1
                else:
                    self.minute = 0
                    if self.hour + 1 < 24:
                        self.hour += 1
                    else:
                        self.hour = 0

    def show(self):
        print("%s:%s:%s" % (self.hour, self.minute, self.second))

if __name__ == "__main__":
    clock = Clock(14, 36, 21)
    clock.run(3000)
    clock.show()
```

### 三、一次性走一秒，在主函数中设置一次走一秒，一秒打印一次最新时间，**推荐**
```python
import time
class Clock(object):
    def __init__(self, hour, minute, second):
        self.hour = hour
        self.minute = minute
        self.second = second

    def run(self):
        self.second += 1
        if self.second == 60:
            self.minute += 1
            self.second = 0
            if self.minute == 60:
                self.minute = 0
                self.hour += 1
                if self.hour == 24:
                    self.hour = 0

    def show(self):
        # 时间格式
        print("%02d:%02d:%02d" % (self.hour, self.minute, self.second))

def main():
    clock = Clock(14, 36, 51)
    while True:
        clock.show()
        clock.run()
        time.sleep(1)

if __name__ == "__main__":
    main()
```
